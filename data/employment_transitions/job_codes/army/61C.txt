Job Code: 61C

Description:
Supervises or performs maintenance on propulsion systems and auxiliary equipment of Army marine vessels.

Must be able to perform the duties of any one of the MOS in the CMF Subfield 653 at the 4 skill level; serves as principal noncommissioned officer of company or as operations sergeant in battalion or comparable unit engaged in operations or maintenance of amphibians, landing, or harbor craft; performs duties of operations sergeant; prepares reports and advises on matters pertaining to marine maintenance and operations; serves as First Sergeant of a company; interprets and supervises execution of company policy and SOP; assists in planning, coordinating, and supervising all activities that support the company mission; advises company commander on all matters concerning enlisted personnel, to include assignments, reassignments, transfers, promotions, granting of passes and leaves, punishments, welfare, privileges, and awards; directs and coordinates company administration; forms unit for drill, ceremonies, and other military formations; receives report of personnel present and absent, and reports number of unauthorized absences; holds NCO call to disseminate instructions and information to subordinate enlisted supervisors; coordinates operation of company food service and supply activities; assists company commander in accomplishing unit training; assists in inspection of organizational activities as prescribed by commander, observes discrepancies, and initiates appropriate corrective action; serves as Chief Instructor in a training facility, Chief Adviser to a reserve component unit, or Chief Adviser to a foreign military unit; provides advice and makes, recommendations to the commander aid staff on all matters pertain to enlisted personnel and their families; accompanies the commander on inspections and visits and at ceremonies; assists in inspection of command activities facilities aid personnel as prescribed by the commander; holds First Sergeants' or Sergeants' Major call to announce information and instructions; insures that newly assigned enlisted personnel are instructed in military courtesy, customs, of the service, and command regulations or policies; provides counsel and guidance to NCOs and other enlisted personnel of the command; inspects duties performed by subordinate NCOs; notes discrepancies and initiates appropriate corrective action; assists in reception of visitors to the command; sits as president or, member of command promotion board for NCOs, as authorized by regulations; performs other duties prescribed by, the commander.

Performs organizational maintenance on propulsion systems and auxiliary equipment of Army marine vessels; diagnoses and troubleshoots malfunctions; operates and performs minor adjustments to items such as generators, motors and vessel electrical systems, ramp mechanisms, winches, pumps and vessel piping systems; performs emergency repairs and improvises with temporary field expedients while at sea; determines extent of maintenance required and makes repairs in accordance with maintenance allocation chart; performs soldering, pipefitting and preservation; paints wood, plastic, and metal surfaces; employs gages, meters, and other tools and devices for organizational maintenance; interprets technical blueprints and publications; performs direct and general support maintenance on propulsion systems and auxiliary equipment of Army marine vessels; diagnoses malfunctions of engine and auxiliary equipment; replaces and repairs engine and auxiliary equipment; tests, repairs, and recharges vessel refrigerant systems; employs hand and power tools, precision tools, oxyacetylene welding and other equipment necessary for direct and general support maintenance.

Must be able to perform the duties of Marine Engineer (61C20 or 61C30); supervises organizational, direct support and general support maintenance on propulsion systems and auxiliary equipment of Army marine vessels; instructs repairmen in proper maintenance procedures and safety techniques; advises on complex repair problems; prepares repair specifications and work orders; applies production and quality control procedures to maintenance operations; prepares reports and determines man-hours, personnel and equipment requirements for repair of propulsion systems and auxiliary equipment of Army marine vessels; must be able to perform the duties of any one of the MOS in the CMF Subfield 653 at skill level 4; serves as principal noncommissioned officer of company or as operations sergeant in battalion or comparable unit engaged in operations or maintenance of amphibians, landing, or harbor craft; performs duties of operations sergeant; prepares reports and advises on matters pertaining to marine maintenance and operations; serves as First Sergeant of a company; interprets and supervises execution of company policy and SOP; assists in planning, coordinating, and supervising all activities that support the company mission; advises company commander on all matters concerning enlisted personnel, to include assignments, reassignments, transfers, promotions, granting of passes and leaves, punishments, welfare, privileges, and awards; directs and coordinates company administration; forms unit for drill, ceremonies, and other military formations; receives report of personnel present and absent, and reports number of unauthorized absences; holds NCO call to disseminate instructions and information to subordinate enlisted supervisors; coordinates operation of company food service and supply activities; assists company commander in accomplishing unit training; assists in inspection of organizational activities as prescribed by commander, observes discrepancies, and initiates appropriate corrective action; serves as Chief Instructor in a training facility, Chief Adviser to a reserve component unit, or Chief Adviser to a foreign military unit.
